100 bool Inventory::removeItemById(const std::string& id) {
101  if (id.empty()) {
102  return false;
103  }
104  auto mapIt = itemsById.find(id);
105  if (mapIt == itemsById.end()) {
106  return false;
107  }
108 
109  StockItem* itemToDelete = mapIt->second;
110  if (itemToDelete == nullptr) {
111  itemsById.erase(mapIt);
112  return false;
113  }
114 
115  const std::string itemName = itemToDelete->getname();
116  PlantInstance* plant = itemToDelete->getplant();
117 
118  auto vecIt = std::find_if(items.begin(), items.end(),
119  [itemToDelete](const std::unique_ptr<StockItem>& item) {
120  return item.get() == itemToDelete;
121  });
122  if (vecIt == items.end()) {
123  itemsById.erase(mapIt);
124  return false;
125  }
126 
127  std::unique_ptr<StockItem> removed = std::move(*vecIt);
128  items.erase(vecIt);
129  itemsById.erase(mapIt);
130  removed.reset();
131  unregisterPlant(plant);
132  if (getStockCount(itemName) == 0) {
133  plantInstanceCounters.erase(plantCounterKey(itemName));
134  }
135  return true;
136 }

224 PlantInstance* Inventory::createPlantInstance(const std::string& plantName) {
225  if (!isPlantType(plantName)) {
226  return nullptr;
227  }
228 
229  std::unique_ptr<Plant> prototype;
230  if (plantRegistry != nullptr) {
231  if (Plant* clone = plantRegistry->createPlant(plantName, "")) {
232  prototype.reset(clone);
233  }
234  }
235 
236  std::unique_ptr<PlantInstance> instance = std::make_unique<PlantInstance>(prototype.get(), "");
237  PlantInstance* rawInstance = instance.get();
238  if (prototype) {
239  if (WaterStrategy* water = prototype->getDefaultWaterStrat()) {
240  rawInstance->setWaterStrategy(water);
241  }
242  if (FertilizeStrategy* fertiliser = prototype->getDefaultFertStrat()) {
243  rawInstance->setFertilizeStrategy(fertiliser);
244  }
245  }
246  if (!prototype) {
247  const std::string counterKey = plantCounterKey(plantName);
248  const int counter = ++plantInstanceCounters[counterKey];
249  const std::string baseName = plantCounterKey(plantName);
250  rawInstance->rename(baseName + std::to_string(counter));
251  }
252 
253  registerPlant(instance.release(), true);
254  if (prototype) {
255  prototypeOwners.emplace(rawInstance, std::move(prototype));
256  }
