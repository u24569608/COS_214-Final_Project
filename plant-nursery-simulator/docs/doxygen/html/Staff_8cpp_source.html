48 void Staff::update(const ObserverEvent& event) {
49  if (event.type == ObserverEventType::SubjectDestroyed) {
50  if (event.source != nullptr) {
51  observedPlants.erase(static_cast<PlantInstance*>(event.source));
52  }
53  if (!event.message.empty()) {
54  careReminders.push_back({StaffReminderType::Message, event.message});
55  }
56  return;
57  }
58 
59  PlantInstance* plant = event.source ? dynamic_cast<PlantInstance*>(event.source) : nullptr;
60  const bool hasTrackedPlants = !observedPlants.empty();
61 
62  if (plant) {
63  if (hasTrackedPlants && observedPlants.count(plant) == 0) {
64  return;
65  }
66  } else if (hasTrackedPlants) {
67  // Ignore broadcasts unrelated to tracked plants.
68  return;
69  }
70 
71  if (event.type == ObserverEventType::CareRequired) {
72  careReminders.push_back({StaffReminderType::Care, buildReminderMessage(event)});
73  return;
74  }
75 
76  if (event.type == ObserverEventType::AvailabilityChanged) {
77  std::string message = event.message;
78  if (message.empty() && event.availability.has_value()) {
79  const bool available = *event.availability == AvailabilityStatus::Available;
80  message = available ? "Plant ready for sale" : "Plant unavailable for sale";
81  }
82  careReminders.push_back({StaffReminderType::Availability,
83  buildReminderMessage({ObserverEventType::CareRequired,
84  event.source,
85  message,
86  event.availability})});
87  return;
88  }
89 }
