262 Order* SalesFacade::buildAndFinalizeOrder(Customer* customer, std::vector<StockItem> items) {
263  // 1. Build the order
264  orderBuilder->createNewOrder();
265  for (StockItem& item : items) {
266  // We pass the address of the item in our local vector copy
267  orderBuilder->addItem(&item);
268  }
269 
270  Order* order = orderBuilder->getOrder();
271  if (order == nullptr) {
272  return nullptr; // Builder failed
273  }
274 
275  // 2. Process Payment
276  double total = order->calculateTotal();
277  // Using a mock string since 'customer' is unused
278  bool paid = paymentProcessor->processPayment("MockCustomerDetails", total);
279 
280  // 3. Finalize and update inventory
281  if (paid) {
282  order->setOrderStatus("Paid");
283  // Remove all items from inventory
284  for (StockItem& item : items) {
285  const std::string itemName = item.getname();
286  StockItem* itemInStock = inventory->findItem(itemName);
287  std::string itemId = itemInStock != nullptr ? itemInStock->getId() : "";
288 
289  if (customer != nullptr) {
290  if (!itemId.empty()) {
291  customer->removeFromCartById(itemId);
292  } else {
293  customer->removeFromCart(itemName);
294  }
295  }
296 
297  bool removed = false;
298  if (!itemId.empty()) {
299  removed = inventory->removeItemById(itemId);
300  } else {
301  const int countBefore = inventory->getStockCount(itemName);
302  inventory->removeItem(itemName);
303  removed = countBefore > inventory->getStockCount(itemName);
304  }
305 
306  if (removed) {
307  notifyItemSold(itemId, itemName);
308  } else {
309  std::cerr << "[SalesFacade] Warning: Unable to retire '" << itemName
310  << "' while finalising order.\n";
311  }
312  }
313  std::cout << "[SalesFacade] Custom order finalized and paid." << std::endl;
314  return order;
315  } else {
316  // Payment failed, we must delete the order
317  std::cout << "[SalesFacade] Custom order failed: Payment declined." << std::endl;
318  delete order;
319  return nullptr;
320  }
321 }
