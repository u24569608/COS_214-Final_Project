60  * @brief Loads inventory from a CSV file and recreates greenhouse-backed plant instances.
61  * @note Stock names with registered prototypes are promoted to plant stock so greenhouse beds stay in sync.
62  */
63 void CSVAdapter::loadInventory(std::string filePath, Inventory* inventory) {
64  if (!inventory) {
65  std::cerr << "[CSVAdapter] Error: Inventory pointer is null." << std::endl;
66  return;
67  }
68 
69  const std::vector<std::vector<std::string>> rows = csvReader->readCsv(filePath);
70 
71  for (size_t i = 0; i < rows.size(); ++i) {
72  const std::vector<std::string>& row = rows[i];
73  if (row.empty()) {
74  continue;
75  }
76 
77  const std::string& entryType = row[0];
78  if (entryType == "Item") {
79  if (row.size() < 3U) {
80  std::cerr << "[CSVAdapter] Warning: Malformed item row on line " << (i + 1)
81  << " in " << filePath << std::endl;
82  continue;
83  }
84  const std::string& name = row[1];
85  const double price = parsePrice(row[2], i + 1, filePath);
86  inventory->additem(std::make_unique<StockItem>(name, price, nullptr));
87  continue;
88  }
89 
90  if (entryType == "Plant") {
91  if (row.size() < 4U) {
92  std::cerr << "[CSVAdapter] Warning: Malformed plant row on line " << (i + 1)
93  << " in " << filePath << std::endl;
94  continue;
95  }
96  const std::string& name = row[1];
97  const double price = parsePrice(row[2], i + 1, filePath);
98  const std::string& stateLabel = row[3];
99 
100  bool treatAsPlant = true;
101  PlantPrototypeRegistry* registry = inventory->getPlantRegistry();
102  if (registry != nullptr && !registry->hasPrototype(name)) {
103  treatAsPlant = false;
104  }
105  if (treatAsPlant) {
106  inventory->registerPlantType(name);
107  }
108  PlantInstance* plant = treatAsPlant ? inventory->createPlantInstance(name) : nullptr;
109  if (plant != nullptr) {
110  if (std::unique_ptr<PlantState> desiredState = makeStateFromLabel(stateLabel)) {
111  plant->setState(std::move(desiredState));
112  }
113  }
114  inventory->additem(std::make_unique<StockItem>(name, price, plant));
115  continue;
116  }
117 
118  std::cerr << "[CSVAdapter] Warning: Unknown entry type '" << entryType
119  << "' on line " << (i + 1) << " in " << filePath << std::endl;
120  }
121  std::cout << "[CSVAdapter] Loaded data from " << filePath << std::endl;
122 }
