55 void PlantInstance::performWater() {
56  if (!plantState) {
57  applyWaterStrategy();
58  return;
59  }

61  setReplayingAction(false);

105 void PlantInstance::setState(std::unique_ptr<PlantState> nextState) {
106  const bool wasAvailable = isAvailableForSale();
107  const std::string previousName = plantState ? plantState->getName() : "";
108  if (!nextState) {
109  return;
110  }
111 
112  plantState = std::move(nextState);
113  const std::string currentName = plantState ? plantState->getName() : "";
114  const bool isAvailable = isAvailableForSale();
115  const bool enteredWithering = plantState && currentName == "Withering";
116 
117  if (currentName != previousName) {
118  if (wasAvailable != isAvailable) {
119  const AvailabilityStatus availabilityStatus =
120  isAvailable ? AvailabilityStatus::Available : AvailabilityStatus::Unavailable;
121  ObserverEvent availabilityEvent{
122  ObserverEventType::AvailabilityChanged,
123  this,
124  isAvailable ? "Plant ready for sale" : "Plant unavailable for sale",
125  std::optional<AvailabilityStatus>{availabilityStatus}};
126  notify(availabilityEvent);
127  }
128 
129  if (enteredWithering) {
130  careAlertActive = true;
131  const ObserverEvent careEvent{
132  ObserverEventType::CareRequired,
133  this,
134  "Plant entered Withering state and needs attention",
135  std::nullopt};
136  notify(careEvent);
137  } else if (!isThirsty() && !needsFertilizing()) {
138  careAlertActive = false;
139  }
140  }

250 void PlantInstance::requestCareIfNeeded() {
251  const bool thirsty = isThirsty();
252  const bool hungry = needsFertilizing();
253  if (!thirsty && !hungry) {
254  careAlertActive = false;

256  }

258  if (careAlertActive) {

263

267  } else if (thirsty) {

271  }

273  const ObserverEvent event{
274  ObserverEventType::CareRequired,
275  this,
276  message,
